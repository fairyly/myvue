```
npm i v-charts echarts -S

```

## issue

>如果 y轴有小数，设置隐藏

```
<ve-histogram
  :data="barChartData"
  :extend="chartExtend"
></ve-histogram>


this.chartExtend = {
  yAxis: {
    axisLabel: {
      formatter (item) {
        if (~String(item).indexOf('.')) return ''
        return item
      }
    }
  }
}
```
